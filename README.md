  onChange,
}) => {
  const [code, setCode] = useState(initialCode);

  const handleChange = useCallback(
    (newCode: string) => {
      setCode(newCode);
      onChange?.(newCode);
    },
    [onChange]
  );

  return (
    <textarea
      value={code}
      onChange={(e) => handleChange(e.target.value)}
      className="font-mono text-sm"
    />
  );
};
